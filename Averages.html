averages = data[:,1]#np.cumsum(rolls) / np.arange(1, 301)

def x_vals(num):
    return list(range(len(averages)))[-num:]

def y_vals(xs):
    na=[]
    print('El promedio de los ultimos '+str(len(xs))+' meses',np.mean(averages[-len(xs):]))

    ax.plot(xs,[np.mean(averages[-len(xs):])]*len(xs),'k--')
    plt.show()
    return averages[-len(xs):]
